#include <string>
#include <vector>
#include <algorithm>

using namespace std;

int solution(vector<vector<int>> dots) {
    int answer = 0;
    sort(dots.begin(), dots.end());
    int width = dots[3][0]-dots[0][0]; // x 가장 큰 값에서 x 가장 작은 값 빼기
    int height = dots[3][1]-dots[0][1]; // y 가장 큰 값에서 y 가장 작은 값 빼기
    answer = width * height;
    
    return answer;
}
